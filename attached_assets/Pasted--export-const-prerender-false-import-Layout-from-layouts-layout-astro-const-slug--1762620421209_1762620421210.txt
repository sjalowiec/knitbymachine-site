export const prerender = false;

import Layout from "../../layouts/layout.astro";

const { slug } = Astro.params;

const GITHUB_TOKEN = import.meta.env.GITHUB_TOKEN || process.env.GITHUB_TOKEN;
const REPO_OWNER = 'sjalowiec';
const REPO_NAME = 'knitbymachine';
const CONTENT_PATH = 'astro-site/src/content';

async function getGlossaryTerm(slug: string) {
  try {
    const response = await fetch(
      `https://api.github.com/repos/${REPO_OWNER}/${REPO_NAME}/contents/${CONTENT_PATH}/glossary-complete.json`,
      {
        headers: {
          'Authorization': `token ${GITHUB_TOKEN}`,
          'Accept': 'application/vnd.github.v3+json',
          'User-Agent': 'knitbymachine-astro'
        }
      }
    );

    if (!response.ok) {
      return null;
    }

    const data = await response.json();
    const content = atob(data.content);
    const allEntries = JSON.parse(content);

    const entry = allEntries.find(e => e.slug === slug);

    if (!entry || entry.isDraft || entry.pendingPublish) {
      return null;
    }

    return entry;
  } catch (error) {
    console.error('Error fetching glossary term:', error);
    return null;
  }
}

const entry = await getGlossaryTerm(slug);

if (!entry) {
  return Astro.redirect('/glossary');
}

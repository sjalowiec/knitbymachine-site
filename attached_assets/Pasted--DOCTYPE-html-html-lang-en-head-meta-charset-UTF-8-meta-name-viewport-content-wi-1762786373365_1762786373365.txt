<script>
  const tabs=document.querySelectorAll('.tab-btn');
  const contents=document.querySelectorAll('.tab-content');
  const triedButtons=document.querySelectorAll('.mark-tried');
  const notes=document.getElementById('notes');

  // restore state
  document.addEventListener('DOMContentLoaded',()=>{
    triedButtons.forEach(btn=>{
      const id=btn.dataset.id;
      if(localStorage.getItem('increase_tried_'+id)==='true'){
        btn.classList.add('done');
        btn.textContent='Tried ✅';
      }
    });
    const savedNotes=localStorage.getItem('increase_notes');
    if(savedNotes) notes.value=savedNotes;
  });

  // tab switching
  tabs.forEach(tab=>{
    tab.addEventListener('click',()=>{
      tabs.forEach(t=>t.classList.remove('active'));
      tab.classList.add('active');
      const target=tab.dataset.tab;
      contents.forEach(c=>c.classList.remove('active'));
      document.getElementById('tab-'+target).classList.add('active');
    });
  });

  // mark as tried
  triedButtons.forEach(btn=>{
    btn.addEventListener('click',()=>{
      btn.classList.toggle('done');
      const done=btn.classList.contains('done');
      btn.textContent=done?'Tried ✅':'Mark as Tried';
      localStorage.setItem('increase_tried_'+btn.dataset.id,done);
    });
  });

  // save notes
  notes.addEventListener('input',()=>localStorage.setItem('increase_notes',notes.value));
</script>
